kind: Deployment
apiVersion: apps/v1
metadata:
  name: webapp-nginx-deployment
  labels:
    app: webapp-nginx
spec:
  replicas: 2
  selector:
    # This must match the labels we set on the pod!
    matchLabels:
      app: webapp-nginx
  
  # This template field is a regular pod configuration 
  # nested inside the deployment spec
  template:
    metadata:
      # Set labels on the pod.
      # This is used in the deployment selector.
      labels:
        app: webapp-nginx
    spec:
      containers:
        - name: webapp-nginx
          image: demoaksacr321210.azurecr.io/webapp
          ports:
            - containerPort: 80

---
apiVersion: v1
kind: Service  
metadata:
  name: webapp-lb-service
  labels:
    app: webapp-nginx
spec:
  type: LoadBalancer
  selector:
    app: webapp-nginx
  ports:
    - port: 80
      targetPort: 80
